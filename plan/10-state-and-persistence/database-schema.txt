================================================================================
DATABASE SCHEMA
================================================================================

Use SQLite for local/MVP. Postgres for team/production.
Schema managed with Alembic (Python) or golang-migrate (Go).

CORE TABLES
--------------------------------------------------------------------------------

tasks
    id              TEXT PRIMARY KEY
    user_id         TEXT NOT NULL
    channel         TEXT                -- "cli" | "web" | "slack" | etc.
    source          TEXT                -- "ui" | "issue" | "loop" | "api"
    source_url      TEXT                -- linked issue URL if any
    repo_url        TEXT NOT NULL
    base_branch     TEXT NOT NULL
    task_text       TEXT NOT NULL
    status          TEXT NOT NULL       -- "pending" | "running" | "done" | "failed" | ...
    subtask_role    TEXT                -- "standalone" | "worker" | "planner" | "reviewer"
    parent_task_id  TEXT REFERENCES tasks(id)
    loop_id         TEXT REFERENCES prompt_loops(id)
    created_at      TIMESTAMP NOT NULL
    updated_at      TIMESTAMP NOT NULL

sessions
    id              TEXT PRIMARY KEY    -- usually = task_id for coding tasks
    task_id         TEXT REFERENCES tasks(id)
    sandbox_id      TEXT
    working_branch  TEXT
    current_node    TEXT
    state_json      TEXT                -- full AgentState as JSON
    trace_id        TEXT
    started_at      TIMESTAMP
    ended_at        TIMESTAMP
    status          TEXT

node_runs
    id              TEXT PRIMARY KEY
    session_id      TEXT REFERENCES sessions(id)
    node_name       TEXT NOT NULL
    node_type       TEXT                -- "deterministic" | "llm_subsession"
    started_at      TIMESTAMP
    ended_at        TIMESTAMP
    status          TEXT                -- "ok" | "warning" | "failed"
    input_state_hash TEXT
    output_state_hash TEXT
    error           TEXT

model_calls
    id              TEXT PRIMARY KEY
    session_id      TEXT REFERENCES sessions(id)
    node_run_id     TEXT REFERENCES node_runs(id)
    provider        TEXT                -- "anthropic" | "openai"
    model           TEXT
    prompt_version  TEXT
    prompt_tokens   INTEGER
    completion_tokens INTEGER
    cost_usd        REAL
    latency_ms      INTEGER
    status          TEXT
    request_artifact_id  TEXT          -- redacted request stored as artifact
    response_artifact_id TEXT          -- redacted response stored as artifact
    created_at      TIMESTAMP

tool_calls
    id              TEXT PRIMARY KEY
    session_id      TEXT REFERENCES sessions(id)
    node_run_id     TEXT REFERENCES node_runs(id)
    tool_name       TEXT NOT NULL
    arguments_json  TEXT
    result_json     TEXT
    status          TEXT
    exit_code       INTEGER
    duration_ms     INTEGER
    requires_approval BOOLEAN
    approval_id     TEXT REFERENCES approvals(id)
    created_at      TIMESTAMP

artifacts
    id              TEXT PRIMARY KEY
    session_id      TEXT REFERENCES sessions(id)
    type            TEXT                -- "diff" | "stdout" | "screenshot" | etc.
    path_or_uri     TEXT NOT NULL       -- local path or S3/GCS URI
    content_hash    TEXT
    size_bytes      INTEGER
    created_at      TIMESTAMP
    metadata_json   TEXT

approvals
    id              TEXT PRIMARY KEY
    session_id      TEXT REFERENCES sessions(id)
    tool_call_id    TEXT
    tool_name       TEXT
    arguments_json  TEXT
    reason          TEXT                -- agent's explanation for why it needs approval
    status          TEXT                -- "pending" | "approved" | "denied" | "expired"
    approver_id     TEXT
    created_at      TIMESTAMP
    decided_at      TIMESTAMP
    expires_at      TIMESTAMP

validation_results
    id              TEXT PRIMARY KEY
    session_id      TEXT REFERENCES sessions(id)
    kind            TEXT                -- "lint" | "test" | "policy" | "ci"
    command         TEXT
    exit_code       INTEGER
    status          TEXT
    summary         TEXT
    artifact_id     TEXT REFERENCES artifacts(id)
    created_at      TIMESTAMP

PROMPT LOOP TABLES
--------------------------------------------------------------------------------

prompt_loops
    id              TEXT PRIMARY KEY
    user_id         TEXT NOT NULL
    name            TEXT NOT NULL
    description     TEXT
    trigger_type    TEXT                -- "cron" | "github_event" | "webhook" | "manual"
    trigger_config  TEXT                -- JSON
    prompt_template TEXT NOT NULL
    target_repo     TEXT NOT NULL
    task_type       TEXT
    max_cost_usd    REAL
    max_wall_seconds INTEGER
    max_runs_per_day INTEGER
    max_concurrent  INTEGER
    output_policy   TEXT
    notify_channel  TEXT
    enabled         BOOLEAN DEFAULT true
    run_count       INTEGER DEFAULT 0
    error_count     INTEGER DEFAULT 0
    last_run_at     TIMESTAMP
    created_at      TIMESTAMP
    updated_at      TIMESTAMP

loop_runs
    id              TEXT PRIMARY KEY
    loop_id         TEXT REFERENCES prompt_loops(id)
    task_id         TEXT REFERENCES tasks(id)
    triggered_by    TEXT                -- "cron" | "github_event" | "manual"
    trigger_event_json TEXT
    started_at      TIMESTAMP
    ended_at        TIMESTAMP
    status          TEXT
    result_url      TEXT
    cost_usd        REAL
    error           TEXT

SUBAGENT TABLES
--------------------------------------------------------------------------------

subtask_specs
    id              TEXT PRIMARY KEY
    parent_task_id  TEXT REFERENCES tasks(id)
    title           TEXT
    description     TEXT
    files_to_modify TEXT                -- JSON array
    files_to_read   TEXT                -- JSON array
    dependencies    TEXT                -- JSON array of subtask spec IDs
    test_hint       TEXT
    priority        INTEGER
    task_id         TEXT REFERENCES tasks(id) -- created when worker is spawned
    status          TEXT                -- "pending" | "enqueued" | "running" | "done" | "failed"

INDEXES
--------------------------------------------------------------------------------

    CREATE INDEX idx_tasks_user_id ON tasks(user_id);
    CREATE INDEX idx_tasks_status ON tasks(status);
    CREATE INDEX idx_tasks_parent ON tasks(parent_task_id);
    CREATE INDEX idx_sessions_task_id ON sessions(task_id);
    CREATE INDEX idx_node_runs_session ON node_runs(session_id);
    CREATE INDEX idx_model_calls_session ON model_calls(session_id);
    CREATE INDEX idx_tool_calls_session ON tool_calls(session_id);
    CREATE INDEX idx_approvals_status ON approvals(status);
    CREATE INDEX idx_loop_runs_loop_id ON loop_runs(loop_id);
    CREATE INDEX idx_subtask_specs_parent ON subtask_specs(parent_task_id);
